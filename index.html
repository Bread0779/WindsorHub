function drawLead(id, sector) { const svg = document.getElementById("leads" + id); svg.innerHTML = ""; const mid = sector.start + (sector.end - sector.start) / 2, rad = (mid - 90) * (Math.PI / 180); const centerX = 160, centerY = 105; const rStart = 85, rExt = 115; const x1 = centerX + rStart * Math.cos(rad), y1 = centerY + rStart * Math.sin(rad); let x2 = centerX + rExt * Math.cos(rad); let y2 = centerY + rExt * Math.sin(rad); // 🛡️ 軸向鎖定核心：高度限位 // 限制引線標籤的高度，使其不超過圓環上下緣 (約 centerY +/- 60px) const yLimit = 60; if (y2 > centerY + yLimit) y2 = centerY + yLimit; if (y2 < centerY - yLimit) y2 = centerY - yLimit; const isRightSide = x2 >= centerX; // 強制橫向拉伸，防止在垂直方向重疊 if (Math.abs(Math.cos(rad)) < 0.5) { x2 = centerX + (isRightSide ? 85 : -85); } const xLabel = isRightSide ? 310 : 10; const yLabel = y2; const anchor = isRightSide ? "end" : "start"; const color = (sector.hasNow && sector.amt < sector.cost) ? "#444" : "#FFF"; const line = document.createElementNS("http://www.w3.org/2000/svg", "polyline"); line.setAttribute("points", `${x1},${y1} ${x2},${y2} ${xLabel},${yLabel}`); line.setAttribute("stroke", color); line.setAttribute("fill", "none"); line.setAttribute("stroke-dasharray", "3,2"); line.setAttribute("stroke-width", "1"); svg.appendChild(line); const t1 = document.createElementNS("http://www.w3.org/2000/svg", "text"); t1.setAttribute("x", xLabel); t1.setAttribute("y", yLabel - 5); t1.setAttribute("fill", "#FFFFFF"); t1.setAttribute("font-size", "12px"); t1.setAttribute("font-weight", "bold"); /* 🛡️ 字體微縮 */ t1.setAttribute("text-anchor", anchor); t1.textContent = sector.name; svg.appendChild(t1); const t2 = document.createElementNS("http://www.w3.org/2000/svg", "text"); const perf = (sector.hasNow && sector.cost > 0) ? `(${((sector.amt-sector.cost)/sector.cost*100).toFixed(1)}%)` : ""; t2.setAttribute("x", xLabel); t2.setAttribute("y", yLabel + 12); t2.setAttribute("fill", sector.amt >= sector.cost ? "#39FF14" : "#FF003C"); t2.setAttribute("font-size", "10px"); t2.setAttribute("font-weight", "bold"); /* 🛡️ 數字微縮 */ t2.setAttribute("text-anchor", anchor); t2.textContent = `$ ${sector.amt.toLocaleString()} ${perf}`; svg.appendChild(t2); }

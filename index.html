function drawLead(id, sector) { const svg = document.getElementById("leads" + id); svg.innerHTML = ""; const mid = sector.start + (sector.end - sector.start) / 2, rad = (mid - 90) * (Math.PI / 180); const centerX = 130, centerY = 105; const r1 = 95; const r2 = 125; const x1 = centerX + r1 * Math.cos(rad), y1 = centerY + r1 * Math.sin(rad); const x2 = centerX + r2 * Math.cos(rad), y2 = centerY + r2 * Math.sin(rad); const isRightSide = x2 >= centerX; const xLabel = isRightSide ? 265 : 5; const yLabel = y2; const color = (sector.hasNow && sector.amt < sector.cost) ? "#222" : "#D4AF37"; const line = document.createElementNS("http://www.w3.org/2000/svg", "polyline"); line.setAttribute("points", `${x1},${y1} ${x2},${y2} ${xLabel},${yLabel}`); line.setAttribute("stroke", color); line.setAttribute("fill", "none"); line.setAttribute("stroke-dasharray", "3,2"); line.setAttribute("stroke-width", "1"); svg.appendChild(line); const t1 = document.createElementNS("http://www.w3.org/2000/svg", "text"); t1.setAttribute("x", xLabel); t1.setAttribute("y", yLabel-8); t1.setAttribute("fill", "#FFFFFF"); t1.setAttribute("font-size", "16px"); t1.setAttribute("font-weight", "bold"); t1.setAttribute("text-anchor", isRightSide?"end":"start"); t1.textContent = sector.name; svg.appendChild(t1); const t2 = document.createElementNS("http://www.w3.org/2000/svg", "text"); const perf = (sector.hasNow && sector.cost > 0) ? `(${((sector.amt-sector.cost)/sector.cost*100).toFixed(1)}%)` : ""; t2.setAttribute("x", xLabel); t2.setAttribute("y", yLabel+16); t2.setAttribute("fill", sector.amt>=sector.cost?"#39FF14":"#FF003C"); t2.setAttribute("font-size", "12px"); t2.setAttribute("font-weight", "bold"); t2.setAttribute("text-anchor", isRightSide?"end":"start"); t2.textContent = `$ ${sector.amt.toLocaleString()} ${perf}`; svg.appendChild(t2); }
